::: center
要居中的段落
:::

::: right
要居右的段落
:::

来对你的段落对齐进行自定义。

# 配置







 





module.exports = {
  plugins: [
    [
      "md-enhance",
      {
        // 启用自定义对齐
        align: true,
      },
    ],
  ],
};
